async function validateTransaction(userId, txid, expectedAmount) {
  // 1. Check TXID uniqueness
  if (await db.txidExists(txid)) {
    throw new Error("TXID already used");

  // 2. Verify on blockchain
  const txInfo = await tronAPI.getTransaction(txid);
  if (!txInfo || txInfo.status !== "SUCCESS") {
    throw new Error("Invalid transaction");

  // 3. Check timing (30 min window)
  const timeDiff = Date.now() - txInfo.timestamp;
  if (timeDiff > 30 * 60 * 1000) {
    throw new Error("Transaction too old");

  // 4. Verify amount
  if (txInfo.amount !== expectedAmount) {
    throw new Error("Amount mismatch");

  // 5. Save to database
  await db.recordTransaction(userId, txid, txInfo);

  return true;
